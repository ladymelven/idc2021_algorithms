Сортировка 2 (результаты прогонов в консоли)

    const maxNumSort = (arr) => {
      let maxIndex = arr.length;
      while (maxIndex > 0) {
        let maxNum = arr[0];
        let maxNumIndex = 0;

        for (let i = 1; i < maxIndex; i++) {
          if (arr[i] > maxNum) {
            maxNum = arr[i];
            maxNumIndex = i;
          }
        }

        arr[maxNumIndex] = arr[maxIndex - 1];
        arr[maxIndex - 1] = maxNum;

        maxIndex--;
      }
      return arr;
    }
