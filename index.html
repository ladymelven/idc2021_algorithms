Пузырьковая сортировка (результаты прогонов в консоли)

    const bubbleSort = (arr) => {
    let maxIndex = arr.length;
    while (maxIndex > 0) {
      for (let i = 1; i < maxIndex; i++) {
        if (arr[i] < arr[i - 1]) {
          const bigger = arr[i - 1];
          arr[i - 1] = arr[i];
          arr[i] = bigger;
        }
      }
      maxIndex--;
    }
    return arr;
  }
